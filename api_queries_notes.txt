# item_keys.json
query MyQuery {
  items(type: keys) {
    id
    name
    normalizedName
    shortName
    description
    minLevelForFlea
    weight
    properties {
      ... on ItemPropertiesKey {
        uses
      }
    }
    wikiLink
    iconLink
    backgroundColor
  }
}
